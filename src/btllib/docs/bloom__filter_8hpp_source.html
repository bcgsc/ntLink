119 inline BloomFilter::BloomFilter(const std::string& path)
120 {
121  std::ifstream file(path);
122 
123  std::string magic_with_brackets =
124  std::string("[") + BLOOM_FILTER_MAGIC_HEADER + "]";
125 
126  std::string line;
127  std::getline(file, line);
128  if (line != magic_with_brackets) {
129  log_error(
130  std::string("Magic string does not match (likely version mismatch)\n") +
131  "Your magic string:\t" + line + "\n" + "BloomFilter magic string:\t" +
132  magic_with_brackets);
133  std::exit(EXIT_FAILURE);
134  }
135 
136  /* Read bloom filter line by line until it sees "[HeaderEnd]"
137  which is used to mark the end of the header section and
138  assigns the header to a char array*/
139  std::string toml_buffer(line + '\n');
140  bool header_end_found = false;
141  while (bool(std::getline(file, line))) {
142  toml_buffer.append(line + '\n');
143  if (line == "[HeaderEnd]") {
144  header_end_found = true;
145  break;
146  }
147  }
148  if (!header_end_found) {
149  log_error("Pre-built bloom filter does not have the correct header end.");
150  std::exit(EXIT_FAILURE);
151  }
152 
153  // Send the char array to a stringstream for the cpptoml parser to parse
154  std::istringstream toml_stream(toml_buffer);
155  cpptoml::parser toml_parser(toml_stream);
156  auto header_config = toml_parser.parse();
157 
158  // Obtain header values from toml parser and assign them to class members
159  auto table = header_config->get_table(BLOOM_FILTER_MAGIC_HEADER);
160  bytes = *table->get_as<size_t>("bytes");
161  hash_num = *table->get_as<unsigned>("hash_num");
162 
163  bytearray.resize(bytes);
164  file.read((char*)bytearray.data(), bytes);
165 }
